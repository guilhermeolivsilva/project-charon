 128    def run(self) -> None:
 129        """Run the program on the virtual machine."""
 130
 131        # Run the instructions related to global vars (that are stored in a
 132        # different section of the program text)
 133        for global_var_instruction in self.program["global_vars"]:
 134            instruction = global_var_instruction["instruction"]
 135            instruction_params = global_var_instruction["metadata"]
 136
 137            instruction_handler = getattr(self, instruction)
 138            instruction_handler(instruction_params)
 139
 140        # Set the `program_counter` to the beginning of the `main` function
 141        try:
 142            self.program_counter = self.program["functions"]["main"]["start"]
 143        except KeyError:
 144            raise SyntaxError("No main function found. Execution aborted.")
 145
 146        # Run the actual program
 147        while True:
 148            code_metadata = self.program["code"][self.program_counter]
 149
 150            instruction = code_metadata[("instruction")]
 151            instruction_params = code_metadata[("metadata")]
 152
 153            if instruction == "HALT":
 154                break
 155
 156            self.program_counter += 1
 157
 158            instruction_handler = getattr(self, instruction)
 159
 160            try:
 161                instruction_handler(instruction_params)
 162            except Exception as e:
 163                print("Bad instruction:", instruction, instruction_params)
 164                raise e

 185    def ADDRESS(self, instruction_params: dict[str, Union[int, float, str]]) -> None:
 186        """
 187        Handle a `ADDRESS` bytecode.
 188
 189        This method loads the address of a variable into a register.
 190
 191        If an array or struct, it will load the address of the first element/
 192        attribute to the register.
 193
 194        Parameters
 195        ----------
 196        instruction_params : dict[str, Union[int, float, str]]
 197            The bytecode metadata.
 198        """
 199
 200        # Value represents the variable to load's ID in the
 201        # source code.
 202        variable_to_load: int = instruction_params["id"]
 203        variable_address: int = int(self.variables[variable_to_load], 16)
 204
 205        offset_size: int = instruction_params.get("offset_size", 0)
 206        offset_register: Union[int, None] = instruction_params.get("offset_register")
 207
 208        offset = offset_size
 209        if offset_register:
 210            offset *= self.registers[offset_register]
 211
 212        variable_address += offset
 213        value: int = instruction_params["register"]
 214        self.registers[value] = hex(variable_address)

 216    def ALLOC(self, instruction_params: dict[str, Union[int, float, str]]) -> None:
 217        """
 218        Handle a `ALLOC` bytecode.
 219
 220        This method allocates memory for a variable.
 221
 222        Parameters
 223        ----------
 224        instruction_params : dict[str, Union[int, float, str]]
 225            The bytecode metadata.
 226
 227        Raises
 228        ------
 229        MemoryError
 230            Raised if the memory is full when this method is called, or if the
 231            available memory is not enough for the variable being allocated.
 232        """
 233
 234        if self.memory_pointer >= self.memory_size:
 235            err_msg: str = "Cannot allocate memory: memory is full"
 236            err_msg += f"\nInstruction: {instruction_params}"
 237            raise MemoryError(err_msg)
 238
 239        variable_address: str = hex(self.memory_pointer)
 240
 241        variable_id: int = instruction_params["id"]
 242        variable_size: int = instruction_params["size"]
 243        variable_address_register: int = instruction_params["register"]
 244
 245        updated_memory_pointer: int = self.memory_pointer + variable_size
 246
 247        if updated_memory_pointer >= self.memory_size:
 248            err_msg: str = "Not enough memory to allocate a new variable."
 249            err_msg += f"\nInstruction: {instruction_params}"
 250            err_msg += f"\nMemory dump:\n{str(self)}"
 251
 252            raise MemoryError(err_msg)
 253
 254        self.registers[variable_address_register] = variable_address
 255        self.variables[variable_id] = variable_address
 256        self.memory_pointer = updated_memory_pointer

 676    def JR(self, instruction_params: dict[str, Union[int, float, str]]) -> None:
 677        """
 678        Handle a `JR` bytecode.
 679
 680        This method jumps to an address stored in a register.
 681
 682        Parameters
 683        ----------
 684        instruction_params : dict[str, Union[int, float, str]]
 685            The bytecode metadata.
 686        """
 687
 688        register_with_address: Union[int, str] = instruction_params["register"]
 689        address_to_jump_to: int = self.registers[register_with_address]
 690
 691        # Handle lists
 692        if isinstance(address_to_jump_to, list):
 693            # Handle the `return` statement from the end of the program: when
 694            # reaching the final `return`, there won't be an address to jump
 695            # to at `registers["ret_address"]`. Thus, jump to the last
 696            # instruction.
 697            if address_to_jump_to:
 698                address_to_jump_to = address_to_jump_to.pop()
 699            else:
 700                address_to_jump_to = len(self.program["code"]) - 1
 701

 129    def run(self) -> None:
 130        """Run the program on the virtual machine."""
 131
 132        # Run the instructions related to global vars (that are stored in a
 133        # different section of the program text)
 134        for global_var_instruction in self.program["global_vars"]:
 135            instruction = global_var_instruction["instruction"]
 136            instruction_params = global_var_instruction["metadata"]
 137
 138            instruction_handler = getattr(self, instruction)
 139            instruction_handler(instruction_params)
 140
 141        # Set the `program_counter` to the beginning of the `main` function
 142        try:
 143            self.program_counter = self.program["functions"]["main"]["start"]
 144        except KeyError:
 145            raise SyntaxError("No main function found. Execution aborted.")
 146
 147        # Run the actual program
 148        while True:
 149            code_metadata = self.program["code"][self.program_counter]
 150
 151            instruction = code_metadata[("instruction")]
 152            instruction_params = code_metadata[("metadata")]
 153
 154            if instruction == "HALT":
 155                break
 156
 157            self.program_counter += 1
 158
 159            instruction_handler = getattr(self, instruction)
 160
 161            try:
 162                instruction_handler(instruction_params)
 163            except Exception as e:
 164                print("Bad instruction:", instruction, instruction_params)
 165                raise e
