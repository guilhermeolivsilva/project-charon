 84    def parse_source_code(self) -> dict[str, dict]:
 85        """
 86        Parse the scopes and functions from the given source code.
 87
 88        Returns
 89        -------
 90        parsed_source_code : dict
 91            A dictionary containing the metadata of the global scope ("globals"
 92            key), and the parsed code of each function ("functions" key).
 93
 94            The `globals` key contains a dictionary such as
 95
 96            {
 97                "variables": {
 98                    <variable name>: <variable type>,
 99                    ...
100                },
101                "structs": {
102                    <struct name>: {
103                        <attribute name>: <attribute type>,
104                        ...
105                    }
106                }
107            }
108
109            where each `scope_name` is either a function or "globals" (for the
110            case of the `global` scope). In this context, the `start` and `end`
111            indices refer to the position of the tokenized source code where
112            the scope lies.
113
114            The `functions` key, on the other hand, contains
115
116            {
117                <function_name>: {
118                    "type": element of `types` or a `struct` from the global
119                            scope,
120                    "parameters": list of (attr_type, attr_name) tuples,
121                    "statements": list of annotated statements found within
122                                  the function.
123                },
124                ...
125            }
126        """
127
128        symbol_collection = self.split_source()
129
130        # Annotate constants with their types
131        symbol_collection = self._annotate_constants(symbol_collection)
132
133        # Parse the global scope first
134        self._parse_globals(symbol_collection)
135
136        # Then, compute the bounding indices of each function, and parse them
137        # one by one.
138        functions_scopes = self._compute_functions_scopes_limits(
139            symbol_collection=symbol_collection
140        )
141
142        # Register the functions and its definitions' ID
143        self.functions = {
144            func_name: {"id": func_id, "prime": func_prime}
145            for func_name, (func_id, func_prime) in zip(
146                functions_scopes.keys(),
147                zip(
148                    range(1, len(functions_scopes) + 1),
149                    primes_list(len(functions_scopes)),
150                ),
151            )
152        }
153
154        for function in functions_scopes:
155            function_data: dict[str, str] = self._parse_function(
156                symbol_collection, **functions_scopes.get(function)
157            )
158
159            # Extend the registered functions with the parsed function data
160            self.functions[function].update(function_data)
161
162        return {"globals": self.globals, "functions": self.functions}

164    def split_source(self) -> list[str]:
165        """
166        Split the source code in individual words and symbols.
167
168        This method is intended to handle reserved words, spaces, line breaks
169        and other style-related issues.
170
171        Returns
172        -------
173        tokenized_source_code : list of str
174            A list of words and individual characters obtained from the source
175            code.
176        """
177
178        # Remove line breaks
179        source_code = self.source_code.replace("\n", "")
180
181        # Replace commas with spaces
182        source_code = source_code.replace(",", " ")
183
184        # Tweak braces, parenthesis and semicolons before splitting the string
185        # by blank spaces
186        tokens_to_tweak = ["(", ")", "{", "}", ";", "[", "]"]
187        for token in tokens_to_tweak:
188            source_code = source_code.replace(token, f" {token} ")
189
190        # TODO: tweak dot operator
191
192        # Split the code word by word (or character by character)
193        source_code = source_code.split(" ")
194
195        # Remove empty tokens
196        remove_empty_filter = (
197            lambda curr_token: curr_token if len(curr_token) > 0 else None
198        )
199        tokenized_source = list(filter(remove_empty_filter, source_code))
200
201        return tokenized_source

203    def _annotate_constants(self, symbol_collection: list[str]) -> list[str]:
204        """
205        Annotate constants with types.
206
207        Parameters
208        ----------
209        symbol_collection : list of str
210            The collection of symbols generated by the `split_source` method.
211
212        Returns
213        -------
214        symbol_collection : list of str
215            The `symbol_collection` with annotated constants.
216        """
217
218        for idx, token in enumerate(symbol_collection):
219            try:
220                _ = int(token)
221                symbol_collection[idx] = "int_cst_" + token
222                continue
223
224            except ValueError:
225                pass
226
227            try:
228                _ = float(token)
229                symbol_collection[idx] = "float_cst_" + token
230                continue
231
232            except ValueError:
233                pass
234
235        return symbol_collection

237    def _parse_globals(self, symbol_collection: list[str]) -> None:
238        """
239        Parse the struct types and variables to be available globally.
240
241        This method fills the `self.globals` dictionary.
242
243        Parameters
244        ----------
245        symbol_collection : list of str
246            The collection of symbols generated by the `split_source` method.
247        """
248
249        curly_brackets_stack: list[int] = []
250
251        for idx, token in enumerate(symbol_collection):
252            if token == "{":
253                curly_brackets_stack.append(1)
254                continue
255
256            elif token == "}":
257                curly_brackets_stack.pop()
258                continue
259
260            # If the `curly_brackets_stack` is empty, then its a global value
261            # We are only interested in structs and variables. Thus, we'll only
262            # handle these constructs for now.
263            if not len(curly_brackets_stack):
264                if token == "struct":
265                    struct_name, struct_attributes = self._handle_struct_definition(
266                        symbol_collection=symbol_collection, struct_idx=idx
267                    )
268
269                    struct_id = len(self.globals["structs"]) + 1
270                    struct_prime = self.struct_prime
271                    self.struct_prime = next_prime(struct_prime)
272
273                    self.globals["structs"][struct_name] = {
274                        "id": struct_id,
275                        "prime": struct_prime,
276                        "attributes": struct_attributes,
277                        "active": False,
278                    }
279
280                elif token in self.globals["structs"]:
281                    continue
282
283                else:
284                    definition_metadata = self._handle_variable_definition(
285                        symbol_collection=symbol_collection,
286                        token_idx=idx,
287                        existing_variables=[*self.globals.get("variables").keys()],
288                    )
289
290                    # The `_handle_variable_definition` method will return an
291                    # empty tuple if the current token is not a variable. Thus,
292                    # only add it to the `globals` dict if the returned value is
293                    # not valid.
294                    if definition_metadata:
295                        variable_name, variable_metadata = definition_metadata
296
297                        var_id = self.variable_count + 1
298                        var_prime = self.variable_prime
299                        variable_metadata["id"] = var_id
300                        variable_metadata["prime"] = var_prime
301                        self.globals["variables"][variable_name] = variable_metadata
302
303                        self.variable_prime = next_prime(self.variable_prime)
304                        self.variable_count += 1

306    def _parse_function(
307        self, symbol_collection: list[str], start_idx: int, end_idx: int
308    ) -> dict[str, str]:
309        """
310        Parse the scope of a function defined in the source code.
311
312        Parameters
313        ----------
314        symbol_collection : list of str
315            The collection of symbols generated by the `split_source` method.
316        start_idx : int
317            The starting index of the function scope in the `symbol_collection`.
318        end_idx : int
319            The ending index of the function scope in the `symbol_collection`.
320
321        Returns
322        -------
323        function_metadata : dict[str, str]
324            The metadata (keys: type, parameters, statements) of the parsed
325            function.
326        """
327
328        # Internal tracking
329        existing_variables: list[str] = [*self.globals.get("variables").keys()]
330
331        # Basic metadata
332        function_name = symbol_collection[start_idx + 1]
333        function_type = symbol_collection[start_idx]
334        parameters = self._extract_parameters(
335            symbol_collection=symbol_collection, function_idx=start_idx
336        )
337
338        # Parse the statements, and initialize the list of statements with a
339        # left curly bracket ({ -- `LCBRA`)
340        statements: list[tuple[str, dict]] = [("LCBRA", {})]
341        statements_start_idx = symbol_collection[start_idx:end_idx].index("{") + 1
342
343        function_symbol_collection = symbol_collection[
344            start_idx + statements_start_idx : end_idx
345        ]
346
347        idx = 0
348
349        # Make it easier to check if a symbol is used without being defined
350        global_variables = deepcopy(self.globals.get("variables"))
351        local_variables = deepcopy(parameters)
352        available_variables = {**global_variables, **local_variables}
353
354        available_types: list[str] = [*self.types, *self.globals["structs"]]
355
356        while idx < len(function_symbol_collection):
357            curr_token = function_symbol_collection[idx]
358
359            # Simply skip the token if it is a known type
360            if curr_token in available_types:
361                idx += 1
362
363            elif "cst" in curr_token:
364                statements.append(("CST", _handle_constant(curr_token)))
365                idx += 1
366
367            elif curr_token in available_variables:
368                previous_token = function_symbol_collection[idx - 1]
369
370                if previous_token in available_types:
371                    raise SyntaxError(f"Redefinition of variable {curr_token}")
372
373                var_metadata = available_variables[curr_token]
374                statements.append(("VAR", var_metadata))
375                idx += 1
376
377            # Handle struct attributes
378            elif "." in curr_token:
379                attribute_access_metadata = self._handle_struct_attribute(
380                    token=curr_token, available_variables=available_variables
381                )
382
383                statements.extend(attribute_access_metadata)
384                idx += 1
385
386            # Handle variables (definition, manipulation) and function calls
387            elif curr_token not in self.reserved_words:
388                # First, check if it is a variable definition.
389                try:
390                    variable_name, variable_type = self._handle_variable_definition(
391                        symbol_collection=function_symbol_collection,
392                        token_idx=idx,
393                        existing_variables=existing_variables,
394                    )
395                    existing_variables.append(variable_name)
396
397                    var_id = self.variable_count + 1
398                    var_prime = self.variable_prime
399
400                    self.variable_count += 1
401                    self.variable_prime = next_prime(self.variable_prime)
402
403                    variable_metadata = {
404                        "name": variable_name,
405                        "id": var_id,
406                        "prime": var_prime,
407                        **variable_type,
408                    }
409
410                    available_variables[variable_name] = variable_metadata
411                    statements.append(("VAR_DEF", variable_metadata))
412
413                # If not, check if it is a function call. If it is not either,
414                # `_handle_function_call` will raise a SyntaxError
415                except (SyntaxError, TypeError):
416                    called_function_name, arguments = self._handle_function_call(
417                        symbol_collection=function_symbol_collection,
418                        available_variables=available_variables,
419                        function_call_idx=idx,
420                    )
421
422                    function_call_metadata = {
423                        "arguments": arguments,
424                        "called_function_metadata": self.functions[
425                            called_function_name
426                        ],
427                    }
428
429                    statements.append(("FUNC_CALL", function_call_metadata))
430
431                    # As the arguments of the function call have already been
432                    # accounted for in `function_call_metadata`, we'll skip
433                    # their tokens. Thus, increment `idx` by 2 (the number of
434                    # parenthesis) plus the number of arguments passed.
435                    idx += 2 + len(function_call_metadata["arguments"])
436
437                idx += 1
438
439            # If not any of the above (but syntatically correct), it is a
440            # known symbol or reserved word of the language
441            else:
442                statements.append((self.reserved_words.get(curr_token), {}))
443                idx += 1
444
445        # Add a closing right curly bracket (} -- `RCBRA`) to the statements
446        # list.
447        statements.append(("RCBRA", {}))
448
449        function_metadata = {
450            "type": function_type,
451            "parameters": parameters,
452            "statements": statements,
453        }
454
455        return function_metadata

457    def _handle_struct_definition(
458        self, symbol_collection: list[str], struct_idx: int
459    ) -> tuple[str, dict[str, str], int]:
460        """
461        Handle a struct type definition.
462
463        The extracted attributes are returned as a dictionary with mappings of
464        `attr_name`: `attr_type`.
465
466        Parameters
467        ----------
468        symbol_collection : list of str
469            The collection of symbols generated by the `split_source` method.
470        struct_idx : int
471            The index of the `struct` keyword in the `symbol_collection` list.
472
473        Returns
474        -------
475        struct_metadata : tuple of (<struct_name>, <attributes>)
476            A tuple containing the name of the struct at the first position,
477            and the dictionary of its attributes (mapping of
478            `attr_name`: `attr_type`).
479
480        Raises
481        ------
482        SyntaxError
483            Raised if
484                - the struct is named after a reserved word or symbol;
485                - any attributes are named after a reserved word or symbol;
486                - an attribute is redefined within the struct;
487                - the struct uses an unknown type for one of its attributes.
488        """
489
490        struct_name: str = symbol_collection[struct_idx + 1]
491        attributes: dict[str, str] = {}
492
493        # Offset of three tokens: `struct`, `<struct name>` and `{`
494        struct_start: int = struct_idx + 3
495        subset: list[str] = symbol_collection[struct_start:]
496
497        idx: int = 0
498
499        # Extract
500        while idx < len(subset):
501            if subset[idx] == "}":
502                break
503
504            attr_type, attr_name = subset[idx : idx + 2]
505
506            if attr_name in attributes:
507                err_msg = (
508                    f"Redefinition of attribute '{attr_name}' in struct"
509                    f" '{struct_name}'"
510                )
511                raise SyntaxError(err_msg)
512
513            is_builtin_type = attr_type in self.types
514            if not is_builtin_type:
515                # Flag the struct type to be `active`, as a variable of its
516                # type is being defined
517                try:
518                    self.globals["structs"][attr_type]["active"] = True
519                except KeyError:
520                    raise SyntaxError(
521                        f"Attribute {attr_name} of struct {struct_name}"
522                        + f" is being declared with unknown type {attr_type}."
523                    )
524
525            attributes[attr_name] = {"type": attr_type, "attr_pointer": len(attributes)}
526
527            # Expected format: `<attr_type>` `<attr_name>` `;`.
528            # Thus, offset 3 tokens
529            idx += 3
530
531        # Validate
532        # 1. Check if the struct has a valid name
533        if struct_name in self.reserved_words:
534            err_msg = f"Invalid struct name '{struct_name}'"
535            raise SyntaxError(err_msg)
536
537        for attr_name, attr_metadata in attributes.items():
538            attr_type = attr_metadata["type"]
539
540            # 2. Check if all the types are valid
541            if attr_type not in self.types or attr_type is None:
542                err_msg = (
543                    f"Unknown type '{attr_type}' of struct attribute" f" '{attr_name}'"
544                )
545                raise SyntaxError(err_msg)
546
547            # 3. Check if there are attributes named after reserved words
548            if attr_name in self.reserved_words:
549                err_msg = (
550                    f"Invalid name of attribute '{attr_name}' in struct"
551                    f" '{struct_name}'"
552                )
553                raise SyntaxError(err_msg)
554
555        struct_metadata = (struct_name, attributes)
556
557        return struct_metadata

559    def _handle_variable_definition(
560        self,
561        symbol_collection: list[str],
562        token_idx: int,
563        existing_variables: list[str],
564    ) -> tuple[str, str]:
565        """
566        Handle a variable definition.
567
568        This method first tests if the token at the given `token_idx` is a
569        variable. If it is, then validate it (check the `Raises` section for
570        more details) and return its name and type. If not, simply return an
571        empty tuple.
572
573        Parameters
574        ----------
575        symbol_collection : list of str
576            The collection of symbols generated by the `split_source` method.
577        token_idx : int
578            The index of the current token in the `symbol_collection` list.
579        existing_variables : list[str]
580            A list containing the names of the currently existing variables.
581            This prevents irrestricted variable redefinition.
582
583        Returns
584        -------
585        : tuple of (`variable_name`, `variable_type`) or empty tuple
586            A tuple of the variable metadata. If not a variable, an empty tuple.
587
588        Raises
589        ------
590        SyntaxError
591            Raised if
592             - the variable is a redefinition;
593             - the variable is of unknown type;
594             - the variable is not followed by a valid token (`;`, `(`, `)`,
595               `=`, `[`, `]`).
596        """
597
598        available_types: list[str] = [*self.types, *self.globals["structs"]]
599
600        try:
601            # First, discard tokens that represent reserved words/symbols or constants
602            variable_name = symbol_collection[token_idx]
603
604            if variable_name in self.reserved_words or "cst" in variable_name:
605                return tuple()
606
607            if variable_name in existing_variables:
608                raise SyntaxError(f"Redefinition of variable {variable_name}")
609
610            previous_token = symbol_collection[token_idx - 1]
611            next_token = symbol_collection[token_idx + 1]
612
613            previous_token_is_valid_type = previous_token in [
614                *available_types,
615                "struct",
616            ]
617            simple_variable_definition = next_token in [";", "="]
618            array_definition = next_token == "["
619
620            variable_type = previous_token
621
622            next_token_is_valid = next_token in [
623                "(",
624                ")",
625                "{",
626                "}",
627                ";",
628                "=",
629                "[",
630                "]",
631                *available_types,
632            ]
633
634            if not next_token_is_valid:
635                err_msg = (
636                    "Syntax error near definition of variable" f" '{variable_name}'"
637                )
638                raise SyntaxError(err_msg)
639
640            if previous_token_is_valid_type:
641                is_builtin_type = variable_type in self.types
642                if not is_builtin_type:
643                    # Flag the struct type to be `active`, as a variable of its
644                    # type is being defined
645                    try:
646                        self.globals["structs"][variable_type]["active"] = True
647                    except (KeyError, IndexError):
648                        raise SyntaxError(
649                            f"Variable {variable_name} is being declared with"
650                            + f" unknown type {variable_type}."
651                        )
652
653                variable_metadata = {"type": variable_type}
654
655                # Export attributes information if it is a struct!
656                if variable_type in self.globals["structs"]:
657                    variable_metadata = {
658                        **variable_metadata,
659                        "attributes": self.globals["structs"][variable_type][
660                            "attributes"
661                        ],
662                    }
663
664                if simple_variable_definition:
665                    return variable_name, variable_metadata
666
667                elif array_definition:
668                    array_length = _handle_constant(
669                        annotated_constant=symbol_collection[token_idx + 2],
670                        number_only=True,
671                    )
672
673                    # Remove the left and right brackets, and the array length
674                    # from the `symbol_collection` so array definitions will
675                    # not be mistaken with array item accesses.
676                    del symbol_collection[token_idx + 1 : token_idx + 4]
677
678                    array_metadata = {**variable_metadata, "length": array_length}
679
680                    return variable_name, array_metadata
681
682            elif not previous_token_is_valid_type:
683                err_msg = (
684                    f"Variable '{variable_name}' of unknown type" f" '{variable_type}'"
685                )
686                raise SyntaxError(err_msg)
687
688        except IndexError:
689            return tuple()

691    def _handle_struct_attribute(
692        self, token: str, available_variables: dict[str, dict[str, str]]
693    ) -> list[str, tuple]:
694        struct_var, struct_attr = token.split(".")
695
696        # Check if the variable has been declared
697        if struct_var not in available_variables:
698            err_msg = f"Invalid access of attribute '{struct_attr}' of unknown"
699            err_msg += f" struct variable '{struct_var}'"
700            raise SyntaxError(err_msg)
701
702        # Check if the struct type has the attribute of interest
703        struct_type = available_variables[struct_var]["type"]
704        if struct_attr not in self.globals["structs"][struct_type]["attributes"]:
705            err_msg = f"Access of unknown attribute '{struct_attr}' of"
706            err_msg += f" variable '{struct_var}'"
707            raise SyntaxError(err_msg)
708
709        var_metadata = available_variables[struct_var]
710        struct_metadata = self.globals["structs"][struct_type]
711
712        struct_attribute_metadata = {**struct_metadata, **var_metadata}
713
714        attr_pointer = list(self.globals["structs"][struct_type]["attributes"]).index(
715            struct_attr
716        )
717
718        return [
719            ("VAR", struct_attribute_metadata),
720            ("DOT", {}),
721            ("CST", {"type": "int", "value": attr_pointer}),
722        ]

724    def _handle_function_call(
725        self,
726        symbol_collection: list[str],
727        available_variables: dict[str, str],
728        function_call_idx: int,
729    ) -> tuple[str, list[dict]]:
730        """
731        Handle a function call to extract the callee and arguments.
732
733        Parameters
734        ----------
735        symbol_collection : list of str
736            The collection of symbols generated by the `split_source` method.
737        available_variables : dict
738            A dictionary containing the variables available at the scope where
739            the function was called.
740        struct_idx : int
741            The index of the function call in the `symbol_collection` list.
742
743        Returns
744        -------
745        function_name : str
746            The name of the called function.
747        arguments : list[dict]
748            List of arguments' metadata.
749
750        Raises
751        ------
752        SyntaxError
753            Raised if the called function is not defined.
754        """
755
756        function_name: str = symbol_collection[function_call_idx]
757
758        if function_name not in self.functions.keys():

850    def _extract_parameters(
851        self, symbol_collection: list[str], function_idx: int
852    ) -> dict[str, str]:
853        """
854        Extract the parameters from a function.
855
856        The extracted parameters are returned as a dictionary of parameters
857        metadata.
858
859        Parameters
860        ----------
861        symbol_collection : list of str
862            The collection of symbols generated by the `split_source` method.
863        function_idx : int
864            The index of the function name in the `symbol_collection` list.
865
866        Returns
867        -------
868        parameters : dict
869            A mapping of `param_name`: `param_type`. If the function takes no
870            parameters, return an empty list.
871
872        Raises
873        ------
874        SyntaxError
875            Raised if
876             - the function has malformed parameters (i.e., missing type and/or
877               name).
878             - a parameter has unknown type.
879        """
880
881        # Align with the index of the left parenthesis (add 2 to offset).
882        curr_idx: int = function_idx + 2
883        parameters: dict[str, str] = {}
884
885        try:
886            while symbol_collection[curr_idx] != ")":
887                curr_token = symbol_collection[curr_idx]
888
889                if curr_token == "(":
890                    curr_idx += 1
891
892                elif curr_token == ")":
893                    break
894
895                else:
896                    param_type = curr_token
897                    param_name = symbol_collection[curr_idx + 1]
898
899                    if param_type not in [*self.types, *self.globals["structs"]]:
900                        function_name = symbol_collection[function_idx]
901                        err_msg = (
902                            f"Unknown type '{param_type}' in definition of "
903                            + f"function '{function_name}'"
904                        )
905                        raise SyntaxError(err_msg)
906
907                    parameter_id = self.variable_count + 1
908                    parameter_prime = self.variable_prime
909
910                    self.variable_count += 1
911                    self.variable_prime = next_prime(self.variable_prime)
912
913                    parameters[param_name] = {
914                        "type": param_type,
915                        "id": parameter_id,
916                        "prime": parameter_prime,
917                    }
918                    curr_idx += 2
919
920        except IndexError:
921            function_name = symbol_collection[function_idx]
922            err_msg = f"Function '{function_name}' has malformed parameters."
923            raise SyntaxError(err_msg)
924
925        return parameters

936            curly_brackets_stack.append(1)

 90        -------
 91        parsed_source_code : dict

312

314        ----------
315        symbol_collection : list of str
316            The collection of symbols generated by the `split_source` method.
317        start_idx : int
318            The starting index of the function scope in the `symbol_collection`.
319        end_idx : int
320            The ending index of the function scope in the `symbol_collection`.

330        existing_variables: list[str] = [*self.globals.get("variables").keys()]

333        function_name = symbol_collection[start_idx + 1]
334        function_type = symbol_collection[start_idx]
335        parameters = self._extract_parameters(
336            symbol_collection=symbol_collection, function_idx=start_idx
337        )
338

340        # left curly bracket ({ -- `LCBRA`)
341        statements: list[tuple[str, dict]] = [("LCBRA", {})]
342        statements_start_idx = symbol_collection[start_idx:end_idx].index("{") + 1
343

345            start_idx + statements_start_idx : end_idx
346        ]
347
348        idx = 0
349
350        # Make it easier to check if a symbol is used without being defined
351        global_variables = deepcopy(self.globals.get("variables"))

353        available_variables = {**global_variables, **local_variables}
354
355        available_types: list[str] = [*self.types, *self.globals["structs"]]
356
357        while idx < len(function_symbol_collection):
358            curr_token = function_symbol_collection[idx]
359
360            # Simply skip the token if it is a known type
361            if curr_token in available_types:
362                idx += 1
363
364            elif "cst" in curr_token:

367

369                previous_token = function_symbol_collection[idx - 1]

375                statements.append(("VAR", var_metadata))
376                idx += 1
377
378            # Handle struct attributes
379            elif "." in curr_token:
380                attribute_access_metadata = self._handle_struct_attribute(

382                )
383
384                statements.extend(attribute_access_metadata)
385                idx += 1
386
387            # Handle variables (definition, manipulation) and function calls
388            elif curr_token not in self.reserved_words:

399                    var_prime = self.variable_prime
400

410

448        statements.append(("RCBRA", {}))
449
450        function_metadata = {
451            "type": function_type,
452            "parameters": parameters,
453            "statements": statements,
454        }

458    def _handle_struct_definition(
459        self, symbol_collection: list[str], struct_idx: int
460    ) -> tuple[str, dict[str, str], int]:
461        """
462        Handle a struct type definition.
463
464        The extracted attributes are returned as a dictionary with mappings of
465        `attr_name`: `attr_type`.
466
467        Parameters
468        ----------
469        symbol_collection : list of str
470            The collection of symbols generated by the `split_source` method.
471        struct_idx : int
472            The index of the `struct` keyword in the `symbol_collection` list.
473
474        Returns
475        -------
476        struct_metadata : tuple of (<struct_name>, <attributes>)
477            A tuple containing the name of the struct at the first position,
478            and the dictionary of its attributes (mapping of
479            `attr_name`: `attr_type`).
480
481        Raises
482        ------
483        SyntaxError
484            Raised if
485                - the struct is named after a reserved word or symbol;
486                - any attributes are named after a reserved word or symbol;
487                - an attribute is redefined within the struct;
488                - the struct uses an unknown type for one of its attributes.
489        """
490
491        struct_name: str = symbol_collection[struct_idx + 1]
492        attributes: dict[str, str] = {}
493
494        # Offset of three tokens: `struct`, `<struct name>` and `{`
495        struct_start: int = struct_idx + 3
496        subset: list[str] = symbol_collection[struct_start:]
497
498        idx: int = 0
499
500        # Extract
501        while idx < len(subset):
502            if subset[idx] == "}":
503                break
504
505            attr_type, attr_name = subset[idx : idx + 2]
506
507            if attr_name in attributes:
508                err_msg = (
509                    f"Redefinition of attribute '{attr_name}' in struct"
510                    f" '{struct_name}'"
511                )
512                raise SyntaxError(err_msg)
513
514            is_builtin_type = attr_type in self.types
515            if not is_builtin_type:
516                # Flag the struct type to be `active`, as a variable of its
517                # type is being defined
518                try:
519                    self.globals["structs"][attr_type]["active"] = True
520                except KeyError:
521                    raise SyntaxError(
522                        f"Attribute {attr_name} of struct {struct_name}"
523                        + f" is being declared with unknown type {attr_type}."
524                    )
525
526            attributes[attr_name] = {"type": attr_type, "attr_pointer": len(attributes)}
527
528            # Expected format: `<attr_type>` `<attr_name>` `;`.
529            # Thus, offset 3 tokens
530            idx += 3
531
532        # Validate
533        # 1. Check if the struct has a valid name
534        if struct_name in self.reserved_words:
535            err_msg = f"Invalid struct name '{struct_name}'"
536            raise SyntaxError(err_msg)
537
538        for attr_name, attr_metadata in attributes.items():
539            attr_type = attr_metadata["type"]
540
541            # 2. Check if all the types are valid
542            if attr_type not in self.types or attr_type is None:
543                err_msg = (
544                    f"Unknown type '{attr_type}' of struct attribute" f" '{attr_name}'"
545                )
546                raise SyntaxError(err_msg)
547
548            # 3. Check if there are attributes named after reserved words
549            if attr_name in self.reserved_words:
550                err_msg = (
551                    f"Invalid name of attribute '{attr_name}' in struct"
552                    f" '{struct_name}'"
553                )
554                raise SyntaxError(err_msg)
555
556        struct_metadata = (struct_name, attributes)
557
558        return struct_metadata

576        symbol_collection : list of str
577            The collection of symbols generated by the `split_source` method.
578        token_idx : int
579            The index of the current token in the `symbol_collection` list.
580        existing_variables : list[str]

586        : tuple of (`variable_name`, `variable_type`) or empty tuple
587            A tuple of the variable metadata. If not a variable, an empty tuple.
588
589        Raises

591        SyntaxError
592            Raised if
593             - the variable is a redefinition;
594             - the variable is of unknown type;
595             - the variable is not followed by a valid token (`;`, `(`, `)`,
596               `=`, `[`, `]`).
597        """
598
599        available_types: list[str] = [*self.types, *self.globals["structs"]]
600
601        try:
602            # First, discard tokens that represent reserved words/symbols or constants
603            variable_name = symbol_collection[token_idx]
604
605            if variable_name in self.reserved_words or "cst" in variable_name:
606                return tuple()
607
608            if variable_name in existing_variables:
609                raise SyntaxError(f"Redefinition of variable {variable_name}")
610
611            previous_token = symbol_collection[token_idx - 1]
612            next_token = symbol_collection[token_idx + 1]
613
614            previous_token_is_valid_type = previous_token in [
615                *available_types,
616                "struct",
617            ]
618            simple_variable_definition = next_token in [";", "="]
619            array_definition = next_token == "["
620
621            variable_type = previous_token
622
623            next_token_is_valid = next_token in [
624                "(",
625                ")",
626                "{",
627                "}",
628                ";",
629                "=",
630                "[",
631                "]",
632                *available_types,
633            ]
634
635            if not next_token_is_valid:
636                err_msg = (
637                    "Syntax error near definition of variable" f" '{variable_name}'"
638                )
639                raise SyntaxError(err_msg)
640
641            if previous_token_is_valid_type:
642                is_builtin_type = variable_type in self.types
643                if not is_builtin_type:
644                    # Flag the struct type to be `active`, as a variable of its
645                    # type is being defined
646                    try:
647                        self.globals["structs"][variable_type]["active"] = True
648                    except (KeyError, IndexError):
649                        raise SyntaxError(
650                            f"Variable {variable_name} is being declared with"
651                            + f" unknown type {variable_type}."
652                        )
653
654                variable_metadata = {"type": variable_type}
655
656                # Export attributes information if it is a struct!
657                if variable_type in self.globals["structs"]:
658                    variable_metadata = {
659                        **variable_metadata,
660                        "attributes": self.globals["structs"][variable_type][
661                            "attributes"
662                        ],
663                    }
664
665                if simple_variable_definition:
666                    return variable_name, variable_metadata
667
668                elif array_definition:
669                    array_length = _handle_constant(
670                        annotated_constant=symbol_collection[token_idx + 2],
671                        number_only=True,
672                    )
673
674                    # Remove the left and right brackets, and the array length
675                    # from the `symbol_collection` so array definitions will
676                    # not be mistaken with array item accesses.
677                    del symbol_collection[token_idx + 1 : token_idx + 4]
678
679                    array_metadata = {**variable_metadata, "length": array_length}
680
681                    return variable_name, array_metadata
682
683            elif not previous_token_is_valid_type:
684                err_msg = (
685                    f"Variable '{variable_name}' of unknown type" f" '{variable_type}'"
686                )
687                raise SyntaxError(err_msg)
688
689        except IndexError:
690            return tuple()

860        Parameters
861        ----------

863            The collection of symbols generated by the `split_source` method.
864        function_idx : int

876            Raised if
877             - the function has malformed parameters (i.e., missing type and/or
878               name).

880        """
881
882        # Align with the index of the left parenthesis (add 2 to offset).
883        curr_idx: int = function_idx + 2

885
886        try:
887            while symbol_collection[curr_idx] != ")":
888                curr_token = symbol_collection[curr_idx]
889
890                if curr_token == "(":
891                    curr_idx += 1
892
893                elif curr_token == ")":
894                    break
895
896                else:
897                    param_type = curr_token
898                    param_name = symbol_collection[curr_idx + 1]
899
900                    if param_type not in [*self.types, *self.globals["structs"]]:
901                        function_name = symbol_collection[function_idx]
902                        err_msg = (
903                            f"Unknown type '{param_type}' in definition of "
904                            + f"function '{function_name}'"
905                        )
906                        raise SyntaxError(err_msg)
907
908                    parameter_id = self.variable_count + 1
909                    parameter_prime = self.variable_prime
910
911                    self.variable_count += 1
912                    self.variable_prime = next_prime(self.variable_prime)
913
914                    parameters[param_name] = {
915                        "type": param_type,
916                        "id": parameter_id,
917                        "prime": parameter_prime,
918                    }
919                    curr_idx += 2
920

922            function_name = symbol_collection[function_idx]
923            err_msg = f"Function '{function_name}' has malformed parameters."
924            raise SyntaxError(err_msg)
